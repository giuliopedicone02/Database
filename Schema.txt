1. Tabella "Aeroporto":
   - ID_Aeroporto (Chiave primaria)
   - Nome_Aeroporto
   - Città
   - Paese
   - Codice_ICAO

2. Tabella "CompagniaAerea":
   - ID_Compagnia (Chiave primaria)
   - Nome_Compagnia
   - Anno_Fondazione
   - Codice_ICAO

3. Tabella "Passeggero":
   - ID_Passeggero (Chiave primaria)
   - Nome
   - Cognome
   - Data_Nascita
   - Numero_Telefono

4. Tabella "Aereo":
   - ID_Aereo (Chiave primaria)
   - Numero_Serie
   - Modello
   - Capacità_Passeggeri
   - Anno_Fabbricazione
   - Compagnia (Chiave esterna referenzia l'entità "CompagniaAerea")

5. Tabella "Volo":
   - ID_Volo (Chiave primaria)
   - Numero_Volo
   - Data_Partenza
   - Ora_Partenza
   - Aeroporto_Partenza (Chiave esterna referenzia l'entità "Aeroporto")
   - Aeroporto_Arrivo (Chiave esterna referenzia l'entità "Aeroporto")
   - ID_Aereo (Chiave esterna referenzia l'entità "Aereo")

6. Tabella "Prenotazione":
   - ID_Prenotazione (Chiave primaria)
   - Codice_Prenotazione (un codice univoco per ogni prenotazione)
   - Data_Prenotazione
   - ID_Passeggero (Chiave esterna referenzia l'entità "Passeggero")
   - ID_Volo (Chiave esterna referenzia l'entità "Volo")
   - ID_Posto (Chiave esterna referenzia l'entità "Posto") 

7. Tabella "Posto":
   - ID_Posto (Chiave primaria)
   - Numero_Posto
   - Classe (es. Economy, Business, First)
   - ID_Aereo (Chiave esterna referenzia l'entità "Aereo")

8. Tabella "Equipaggio":
   - ID_Equipaggio (Chiave primaria)
   - Nome
   - Cognome
   - Data_Nascita
   - Ruolo (es. Pilota, Assistente di volo, Ecc.)

9. Tabella "AssegnazioneEquipaggio":
   - ID_AssegnazioneEquipaggio (Chiave primaria)
   - ID_Equipaggio (Chiave esterna referenzia l'entità "Equipaggio")
   - ID_Volo (Chiave esterna referenzia l'entità "Volo")

10. Tabella "Servizio":
    - ID_ServizioVolo (Chiave primaria)
    - Descrizione
    - Costo
    - ID_Volo (Chiave esterna referenzia l'entità "Volo")

--- CREAZIONE TABELLE ---

CREATE TABLE Aeroporto (
    ID_Aeroporto INT PRIMARY KEY,
    Nome_Aeroporto VARCHAR(255),
    Città VARCHAR(255),
    Paese VARCHAR(255),
    Codice_ICAO VARCHAR(10)
);


CREATE TABLE CompagniaAerea (
    ID_Compagnia INT PRIMARY KEY,
    Nome_Compagnia VARCHAR(255),
    Anno_Fondazione INT,
    Codice_ICAO VARCHAR(10)
);

CREATE TABLE Passeggero (
    ID_Passeggero INT PRIMARY KEY,
    Nome VARCHAR(255),
    Cognome VARCHAR(255),
    Data_Nascita DATE,
    Numero_Telefono VARCHAR(20)
);

CREATE TABLE Aereo (
    ID_Aereo INT PRIMARY KEY,
    Numero_Serie VARCHAR(50),
    Modello VARCHAR(255),
    Capacità_Passeggeri INT,
    Anno_Fabbricazione INT,
    Compagnia INT,
    FOREIGN KEY (Compagnia) REFERENCES CompagniaAerea (ID_Compagnia)
);

CREATE TABLE Volo (
    ID_Volo INT PRIMARY KEY,
    Numero_Volo VARCHAR(50),
    Data_Partenza DATE,
    Ora_Partenza TIME,
    Aeroporto_Partenza INT,
    Aeroporto_Arrivo INT,
    ID_Aereo INT,
    FOREIGN KEY (Aeroporto_Partenza) REFERENCES Aeroporto (ID_Aeroporto),
    FOREIGN KEY (Aeroporto_Arrivo) REFERENCES Aeroporto (ID_Aeroporto),
    FOREIGN KEY (ID_Aereo) REFERENCES Aereo (ID_Aereo)
);

CREATE TABLE Posto (
    ID_Posto INT PRIMARY KEY,
    Numero_Posto VARCHAR(10),
    Classe VARCHAR(50),
    ID_Aereo INT,
    FOREIGN KEY (ID_Aereo) REFERENCES Aereo (ID_Aereo)
);

CREATE TABLE Equipaggio (
    ID_Equipaggio INT PRIMARY KEY,
    Nome VARCHAR(255),
    Cognome VARCHAR(255),
    Data_Nascita DATE,
    Ruolo VARCHAR(50)
);

CREATE TABLE Servizio (
    ID_ServizioVolo INT PRIMARY KEY,
    Descrizione VARCHAR(255),
    Costo INT,
    ID_Volo INT,
    FOREIGN KEY (ID_Volo) REFERENCES Volo (ID_Volo)
);

CREATE TABLE AssegnazioneEquipaggio (
    ID_AssegnazioneEquipaggio INT PRIMARY KEY,
    ID_Equipaggio INT,
    ID_Volo INT,
    FOREIGN KEY (ID_Equipaggio) REFERENCES Equipaggio (ID_Equipaggio),
    FOREIGN KEY (ID_Volo) REFERENCES Volo (ID_Volo)
);

CREATE TABLE Prenotazione (
    ID_Prenotazione INT PRIMARY KEY,
    Codice_Prenotazione VARCHAR(50) UNIQUE,
    Data_Prenotazione DATE,
    ID_Passeggero INT,
    ID_Volo INT,
    ID_Posto INT,
    FOREIGN KEY (ID_Passeggero) REFERENCES Passeggero (ID_Passeggero),
    FOREIGN KEY (ID_Volo) REFERENCES Volo (ID_Volo),
    FOREIGN KEY (ID_Posto) REFERENCES Posto (ID_Posto)
);
